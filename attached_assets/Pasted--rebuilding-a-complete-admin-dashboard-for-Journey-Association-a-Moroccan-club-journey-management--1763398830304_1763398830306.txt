Step 1.3: Protected Route Component
Create: src/components/admin/ProtectedRoute.tsx
Build route protection that:

Checks if user is authenticated
Verifies admin role (isAdmin === true)
Redirects to /admin/login if not authorized
Shows loading spinner during auth check
Uses React Query to fetch current user

typescript// Example structure
const ProtectedRoute = ({ children }) => {
  const { data: user, isLoading } = useQuery('currentUser', fetchCurrentUser);
  
  if (isLoading) return <LoadingSpinner />;
  if (!user || !user.isAdmin) {
    return <Navigate to="/admin/login" replace />;
  }
  
  return <AdminLayout>{children}</AdminLayout>;
};

typescriptconst BatchProcessor = ({ records, processFn, onComplete }) => {
  const [progress, setProgress] = useState(0);
  const [isPaused, setIsPaused] = useState(false);
  const [results, setResults] = useState({ success: 0, failed: 0 });
  
  const processBatch = async (batch) => {
    for (const record of batch) {
      if (isPaused) break;
      try {
        await processFn(record);
        setResults(r => ({ ...r, success: r.success + 1 }));
      } catch (error) {
        setResults(r => ({ ...r, failed: r.failed + 1 }));
      }
      setProgress(p => p + 1);
    }
  };
  
  // Process in batches of 100
  const batchSize = 100;
  for (let i = 0; i < records.length; i += batchSize) {
    const batch = records.slice(i, i + batchSize);
    await processBatch(batch);
  }
  
  onComplete(results);
};

typescript// src/components/admin/ErrorBoundary.tsx
import { Component, ErrorInfo, ReactNode } from 'react';
import * as Sentry from '@sentry/react';

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    Sentry.captureException(error, {
      contexts: {
        react: {
          componentStack: errorInfo.componentStack,
        },
      },
    });
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback || (
        <div className="flex min-h-screen items-center justify-center p-4">
          <Card className="max-w-md">
            <CardHeader>

Step 1.3: Protected Route Component
Create: src/components/admin/ProtectedRoute.tsx
Build route protection that:

Checks if user is authenticated
Verifies admin role (isAdmin === true)
Redirects to /admin/login if not authorized
Shows loading spinner during auth check
Uses React Query to fetch current user

typescript// Example structure
const ProtectedRoute = ({ children }) => {
  const { data: user, isLoading } = useQuery('currentUser', fetchCurrentUser);
  
  if (isLoading) return <LoadingSpinner />;
  if (!user || !user.isAdmin) {
    return <Navigate to="/admin/login" replace />;
  }
  
  return <AdminLayout>{children}</AdminLayout>;
};
